Let's look at boost::split which you can use to tokenize a std::string

#include<iostream>
#include<boost/algorithm/string.hpp>
int main(int argc,char**argv){
  std::string a("lorem|ipsum|sit|consecutar");
  std::vector<std::string> b;
  boost::split(
    b,
    a,
    boost::is_any_of("|")
  );
  for(auto c:b){
    std::cout<<c<<std::endl;
  }
  return 0;
}
