Let's look at using varargs in c using stdarg.h which provides va_list type and va_start(), va_arg() and va_end().

#include<stdarg.h>
#include<stdio.h>
int maxof(int n_args,...){
	va_list ap;
	va_start(ap,n_args);
	int max=va_arg(ap,int);
	for(int i=2;i<=n_args;i++){
		int a=va_arg(ap,int);
		if(a>max)max=a;
	}
	va_end(ap);
	return max;
}
int main(int argc,char** argv){
	printf("%d\n",maxof(1,2));
	printf("%d\n",maxof(1,2,3));
	printf("%d\n",maxof(4,3,2,1));
	return 0;
}
