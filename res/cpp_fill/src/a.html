STL fill assigns values to elements of a container.

#include<iostream>
#include<vector>
#include<algorithm>
int main(int argc,char** argv){
  std::vector<int> a;
  a.resize(8);
  std::fill(a.begin(),a.end(),0);
  for(auto b:a){
    std::cout<<b<<" ";
  }
  std::cout<<std::endl;
  return 0;
}

Related to std::fill is std::fill_n, which fills the first n elements in the container with a given value, for example

#include<iostream>
#include<vector>
#include<algorithm>
int main(int argc,char** argv){
  std::vector<int> a;
  a.resize(8);
  std::fill(a.begin(),a.end(),0);
  std::fill_n(a.begin(),a.size()/2,1);
  for(auto b:a){
    std::cout<<b<<" ";
  }
  std::cout<<std::endl;
  return 0;
}
