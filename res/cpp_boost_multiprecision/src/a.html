Let's look at boost::multiprecision.

cpp_int

Including boost/multiprecision/cpp_int.hpp you get int128_t, int256_t, int512_t and int1024_t. Using these you can achieve integer precision up to 1024 bits

#include<iostream>
#include<boost/multiprecision/cpp_int.hpp>
int main(int argc,char** argv){
  boost::multiprecision::int128_t a{1234567890};
  boost::multiprecision::int128_t b{1234567890};
  std::cout<<(a*b)<<std::endl;
  return 0;
}

Arbitrary precision is supported by boost::multiprecision::cpp_int if you are unsure about future requirements for integer precision. Precision is computed at run-time

#include<iostream>
#include<boost/multiprecision/cpp_int.hpp>
int main(int argc,char** argv){
  boost::multiprecision::cpp_int a{1234567890};
  for(int i=0;i<8;i++){a*=a;}
  std::cout<<a<<std::endl;
  return 0;
}

boost/multiprecision/cpp_dec_float.hpp contains definitions for boost::multiprecision::cpp_dec_float_50 and boost::multiprecision::cpp_dec_float_100 which offers precision up to 50 and 100 decimal precision respectively. Constants from are templated to specify the type you want to get, e.g. a cpp_dec_float50 pi or a cpp_dec_float100 pi as appropriate.

#include<iostream>
#include<boost/multiprecision/cpp_dec_float.hpp>
#include<boost/math/constants/constants.hpp>
int main(int argc,char** argv){
  boost::multiprecision::cpp_dec_float_50 r{1};
  auto a=2*boost::math::constants::pi<boost::multiprecision::cpp_dec_float_50>()*r;
  std::cout
    <<std::setprecision(std::numeric_limits<boost::multiprecision::cpp_dec_float_50>::digits10)
    <<a
    <<std::endl
  ;
  return 0;
}
