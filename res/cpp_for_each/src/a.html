std::for_each applies a given function object to the result, dereferencing each iterator in the specified range.

#include<iostream>
#include<vector>
void inc(int& a){
	++a;
}
void dec(int& a){
	--a;
}
int main(int argc,char** argv){
	std::vector<int> v={0,1,2,3};
	for(auto a:v){
		std::cout<<a<<" ";
	}
	std::cout<<std::endl;
	std::for_each(v.begin(),v.end(),inc);
	for(auto a:v){
		std::cout<<a<<" ";
	}
	std::cout<<std::endl;
	std::for_each(v.begin(),v.end(),dec);
	for(auto a:v){
		std::cout<<a<<" ";
	}
	std::cout<<std::endl;
	return 0;
}

Here the same, using lambdas

#include<iostream>
#include<vector>
int main(int argc,char** argv){
	std::vector<int> v={0,1,2,3};
	for(auto a:v){
		std::cout<<a<<" ";
	}
	std::cout<<std::endl;
	std::for_each(v.begin(),v.end(),[](int& a){
		a++;
	});
	for(auto a:v){
		std::cout<<a<<" ";
	}
	std::cout<<std::endl;
	std::for_each(v.begin(),v.end(),[](int& a){
		a--;
	});
	for(auto a:v){
		std::cout<<a<<" ";
	}
	std::cout<<std::endl;
	return 0;
}

You dont have to change the elements being iterated, for example

std::for_each(v.begin(),v.end(),[](int& a){
	std::cout<<a<<std::endl;
});

You can also use capturing lambdas to manipulate other containers

std::vector<int> v2;
std::for_each(v.begin(),v.end(),[&](int& a){
	v2.push_back(a*2);
});
for(auto a:v2){
	std::cout<<a<<" "<<std::endl;
}
std::cout<<std::endl;
