STL std::find searches a container for a value and returns an iterator that is either the element matched if it was found or end() if it was not found

#include<iostream>
#include<vector>
#include<algorithm>
int main(int argc,char** argv){
  std::vector v={0,1,2,3};
  std::vector<int>::iterator i;
  {
    i=std::find(v.begin(),v.end(),0);
    if(i!=v.end()){
      std::cout<<"found: ["<<(i-v.begin())<<"]"<<std::endl;
    }else{
      std::cout<<"not found"<<std::endl;
    }
  }
  {
    i=std::find(v.begin(),v.end(),32);
    if(i!=v.end()){
      std::cout<<"found: ["<<(i-v.begin())<<"]"<<std::endl;
    }else{
      std::cout<<"not found"<<std::endl;
    }
  }
  return 0;
}
