c++ type traits provide a way of providing behavior based on the type specifics. For example, based on if an argument to a template function is a pointer or non-pointer type.

#include<iostream>
#include<type_traits>
int main(int argc,char** argv){
	{
		int*a;
		if(std::is_pointer<decltype(a)>()){
			std::cout<<"Pointer"<<std::endl;
		}else{
			std::cout<<"Not Pointer"<<std::endl;
		}
	}
	{
		int a;
		if(std::is_integral<decltype(a)>()){
			std::cout<<"Is Integral"<<std::endl;
		}else{
			std::cout<<"Is not Integral"<<std::endl;
		}
	}
	return 0;
}
