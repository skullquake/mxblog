Update msys2 

pacman -Syu

Then close the terminal and run mingw64.exe 

pacman \
	-S\
	development\
	mingw-w64-x86_64_go\
	base-devel\
	gcc\
	pv\
	python2-pip\
	python3-pip\
	gnu-netcat\
	tmux\
	unzip\
	mc\
	vim\
	cmake\
	mingw-w64-x86_64-nodejs\
	git\
	mingw-w64-x86_64-editrights\
	cygrunsrv\
	openssh

Set up ssh 

ssh-keygen -f /etc/ssh/ssh_host_rsa_key -N '' -t rsa
ssh-keygen -f /etc/ssh/ssh_host_dsa_key -N '' -t dsa
ssh-keygen -f /etc/ssh/ssh_host_ecdsa_key -N '' -t ecdsa
ssh-keygen -f /etc/ssh/ssh_host_ed25519_key -N '' -t ed25519

Prior to starting the ssh server, edit /etc/ssh/sshd_config 

...
UsePrivilegeSeperation no
PasswordAuthentication yes

To start ssh server 

/usr/bin/sshd

Get openjdk 

wget https://download.java.net/java/GA/jdk13/5b8a42f3905b406298b72d750b6919f6/33/GPL/openjdk-13_windows-x64_bin.zip
unzip openjdk-13_windows-x64_bin.zip -d /opt

Install maven

wget https://apache.is.co.za/maven/maven-3/3.6.3/binaries/apache-maven-3.6.2.tar.gz
tar xf apache-maven-3.6.2.tar.gz -C /opt

Edit your ~/.profile

export M2_HOME="/opt/apache-maven-3.6.2"
PATH=$PATH:/opt/apache-maven-3.6.2/bin:/opt/jdk-13/bin
export JAVA_HOME="/opt/jdk-13"
